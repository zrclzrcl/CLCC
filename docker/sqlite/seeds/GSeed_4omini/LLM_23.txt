CREATE TABLE employees(id INTEGER PRIMARY KEY, name TEXT NOT NULL, age INTEGER CHECK (age >= 18), salary REAL);

CREATE TABLE departments(id INTEGER PRIMARY KEY, dept_name TEXT UNIQUE);

CREATE TABLE emp_dept(emp_id INTEGER, dept_id INTEGER, FKEY (emp_id) REFERENCES employees(id), FKEY (dept_id) REFERENCES departments(id));

CREATE INDEX idx_emp_name ON employees(name);

CREATE TEMP TABLE temp_salaries AS SELECT AVG(salary) AS avg_salary FROM employees;

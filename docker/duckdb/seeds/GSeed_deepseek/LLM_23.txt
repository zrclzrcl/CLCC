CREATE TABLE employees (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    salary NUMERIC(10, 2),
    hire_date DATE
);

CREATE TABLE departments (
    dept_id INTEGER PRIMARY KEY,
    dept_name TEXT NOT NULL
);

SELECT e.name AS employee_name, d.dept_name, e.salary, e.hire_date 
FROM employees e 
JOIN departments d ON e.id = d.dept_id;

CREATE TABLE salary_report AS
SELECT 
    e.name,
    e.salary,
    ROW_NUMBER() OVER (ORDER BY e.salary DESC) AS salary_rank
FROM employees e;

ALTER TABLE departments ADD COLUMN location TEXT;

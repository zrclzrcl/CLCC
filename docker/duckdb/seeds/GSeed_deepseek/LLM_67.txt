DROP TABLE IF EXISTS t1;
CREATE TABLE t1(
    id INTEGER PRIMARY KEY,
    name TEXT,
    value REAL,
    is_active BOOLEAN,
    created_at TIMESTAMP,
    data BLOB
);

SELECT * FROM t1;
SELECT id, name, value, is_active 
FROM t1 
WHERE created_at > '2023-01-01 00:00:00' AND data IS NOT NULL
ORDER BY id DESC;

SELECT 
    id,
    name,
    value,
    ROW_NUMBER() OVER (ORDER BY value DESC) AS rank
FROM t1
WHERE is_active = TRUE;

WITH cte AS (
    SELECT id, name, value FROM t1 WHERE value > 50
)
SELECT * FROM cte UNION ALL SELECT * FROM t1 WHERE value <= 50;

RECURSIVE rcte(n) AS (
    VALUES(1)
    UNION ALL
    SELECT n + 1 FROM rcte WHERE n < 10
)

CREATE INDEX idx_test_types ON test_types (id, name);

WITH cte AS (
    SELECT 
        id, 
        name,
        created_at
    FROM 
        test_types
)
SELECT 
    t.id,
    t.name,
    s.content,
    ROW_NUMBER() OVER (ORDER BY t.created_at) as row_num
FROM 
    test_types t
LEFT JOIN 
    test_string s ON t.id = s.id;

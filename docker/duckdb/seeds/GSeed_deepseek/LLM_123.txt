CREATE TABLE t0 (
    c0 INTEGER PRIMARY KEY,
    c1 TEXT,
    c2 REAL,
    c3 BLOB
);

INSERT INTO t0 (c0, c1, c2, c3) VALUES
(1, 'test', 3.14, X'0102'),
(2, 'example', 42.0, X'0304');

SELECT 
    c0,
    c1,
    c2,
    c3,
    ROW_NUMBER() OVER () AS row_num,
    RANK() OVER (ORDER BY c2 DESC) AS rank_val
FROM t0
WHERE EXISTS (
    SELECT 1 FROM t0 WHERE c0 = t0.c0 AND c2 > 1.0
)
GROUP BY c0, c1, c2, c3;

CREATE INDEX idx_c1 ON t0(c1);

UPDATE t0 SET c2 = 2.718 WHERE c1 LIKE 'te%';

DELETE FROM t0 WHERE c3 = X'0102';
